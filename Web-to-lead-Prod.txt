        function getEstados() {
            let url = 'https://servicodados.ibge.gov.br/api/v1/localidades/estados'
            var list = document.getElementById('estados');

            fetch(url).then(response => response.json())
                .then((json) => {
                    for (let i = 0; i < json.length; i++) {
                        var novaOption = document.createElement("option");
                        novaOption.text = json[i].nome;
                        novaOption.value = json[i].nome;
                        novaOption.setAttribute("data-id", json[i].id)
                        list.options.add(novaOption);
                    }
                })
                .catch((erro) => {
                    let div = document.getElementById("erro");
                    div.innerHTML = 'Página inexistente';
                });
        }

        document.addEventListener("DOMContentLoaded", (event) => {
            var select = document.getElementById('estados');
            var cidadeSelect = document.getElementById('cidades');
            cidadeSelect.disabled = true;

            select.addEventListener('change', function() {
                var option = this.selectedOptions[0];
                var texto = option.textContent;

                cidadeSelect.innerHTML = '<option value=" ">--</option>';
                cidadeSelect.disabled = true;

                var estadoId = option.getAttribute('data-id');

                if (texto != '--') {
                    cidadeSelect.disabled = false;
                    let url = `https://servicodados.ibge.gov.br/api/v1/localidades/estados/${estadoId}/municipios`;
                    var list = document.getElementById('cidades');

                    fetch(url).then(response => response.json())
                        .then((json) => {
                            for (let i = 0; i < json.length; i++) {
                                var novaOption = document.createElement("option");
                                novaOption.text = json[i].nome;
                                novaOption.value = json[i].nome;
                                list.options.add(novaOption);
                            }
                        })
                        .catch((erro) => {
                            let div = document.getElementById("erro");
                            div.innerHTML = 'Página inexistente';
                        });
                }
            });
        });
    </script>

        <script>
            const availableProducts = document.getElementById('availableProducts');
            const selectedProducts = document.getElementById('selectedProducts');
            const addProductButton = document.getElementById('addProduct');
            const removeProductButton = document.getElementById('removeProduct');
    
            
            // Função para mover opções selecionadas para a lista de produtos selecionados
            addProductButton.addEventListener('click', () => {
                const selectedOptions = [...availableProducts.selectedOptions];
                selectedOptions.forEach(option => {
                    selectedProducts.appendChild(option);
                    option.selected = true; // Seleciona o item movido
                    
                });
               
            });
    
            // Função para mover opções selecionadas de volta para a lista de produtos disponíveis
            removeProductButton.addEventListener('click', () => {
                const selectedOptions = [...selectedProducts.selectedOptions];
                selectedOptions.forEach(option => {
                    availableProducts.appendChild(option);
                    option.selected = false; // Desmarca o item movido
                   
                });
                
            });
        </script>
